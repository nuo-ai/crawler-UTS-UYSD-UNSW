# 支持的搜索参数：
# suburb: 区域名称
# price: 价格范围，如 price=0-500
# bedrooms: 卧室数量，如 bedrooms=2-any
# bathrooms: 浴室数量，如 bathrooms=1-any
# carspaces: 车位数量，如 carspaces=1-any
# property-type: 房产类型，如 property-type=apartment,house
